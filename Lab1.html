  1. The aesthetics of the vizualization are the following:

x-asis: income

y-axis: life expectancy

color: world region

size: population

import pandas as pd

df1 = pd.read_csv("/Users/maileyamasaki/Desktop/Cal Poly SLO/Machine Learning/Week_1/q1data.csv")
from plotnine import ggplot, aes, geom_point, theme
from plotnine import *

(ggplot(df1,
aes(
  x = "income",
  y = "life_exp",
  color = "world_bank_region",
  size = "population"
))
+ geom_point()
+ annotate(geom="text", label = "2010")
+ labs(x="Income", y="Life Expectancy", )
+ theme(figure_size=(9, 6)) 
)
